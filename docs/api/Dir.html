def chdir(path, &block)
  if block
    wd = self.getwd
    begin
      self._chdir(path)
      block.call(path)
    ensure
      self._chdir(wd)
    end
  else
    self._chdir(path)
  end
end

def children(path)
  a = []
  self.open(path) do |d|
    while s = d.read
      a << s unless s == "." || s == ".."
    end
  end
  a
end

    d.each(&block)
  end

def open(path, &block)
  if block
    d = self.new(path)
    begin
      block.call(d)
    ensure
      begin
        d.close
      rescue IOError
      end
    end
  else
    self.new(path)
  end
end
