def each(&block)
  return to_enum unless block_given?

  @enums.each { |e| e.each(&block) }

  self
end

def rewind

  end

end

def size
  @enums.reduce(0) do |a, e|
    return nil unless e.respond_to?(:size)
    a + e.size
  end
end
