def >> other
  n = self.to_i
  other = other.to_i
  if other < 0
    n << -other
  else
    other.times { n /= 2 }
    if n.abs < 1
      if n >= 0
        0
      else
        -1
      end
    else
      n.to_i
    end
  end
end
