def clamp(min, max)
  if (min <=> max) > 0
    raise ArgumentError, "min argument must be smaller than max argument"
  end
  c = self <=> min
  if c == 0
    return self
  elsif c < 0
    return min
  end
  c = self <=> max
  if c > 0
    return max
  else
    return self
  end
end
