562 #ifndef ISPRINT
563 #define ISASCII(c) ((unsigned)(c) <= 0x7f)
564 #define ISPRINT(c) (((unsigned)(c) - 0x20) < 0x5f)
565 #define ISSPACE(c) ((c) == ' ' || (unsigned)(c) - '\t' < 5)
566 #define ISUPPER(c) (((unsigned)(c) - 'A') < 26)
567 #define ISLOWER(c) (((unsigned)(c) - 'a') < 26)
568 #define ISALPHA(c) ((((unsigned)(c) | 0x20) - 'a') < 26)
569 #define ISDIGIT(c) (((unsigned)(c) - '0') < 10)
570 #define ISXDIGIT(c) (ISDIGIT(c) || ((unsigned)(c) | 0x20) - 'a' < 6)
571 #define ISALNUM(c) (ISALPHA(c) || ISDIGIT(c))
572 #define ISBLANK(c) ((c) == ' ' || (c) == '\t')
573 #define ISCNTRL(c) ((unsigned)(c) < 0x20 || (c) == 0x7f)
574 #define TOUPPER(c) (ISLOWER(c) ? ((c) & 0x5f) : (c))
